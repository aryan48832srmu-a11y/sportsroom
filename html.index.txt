<script>
    // Initial Inventory Data
    let inventory = [
        { name: "Table Tennis Bat", total: 4, issued: 1 },
        { name: "Football", total: 2, issued: 0 },
        { name: "Pool Cue", total: 4, issued: 2 },
        { name: "Carrom Board", total: 2, issued: 1 },
        { name: "BasketBall", total: 3, issued:1 }
    ];

    // Render Table
    function renderTable() {
        const tbody = document.getElementById('inventory-body');
        tbody.innerHTML = inventory.map(item => `
            <tr>
                <td><b>${item.name}</b></td>
                <td>${item.total}</td>
                <td><span class="count-badge avail">${item.total - item.issued}</span></td>
                <td><span class="count-badge issued">${item.issued}</span></td>
            </tr>
        `).join('');
    }
    renderTable();

    // --- QR Scanner ---
    function onScanSuccess(decodedText) {
        document.getElementById('scan-popup').style.display = 'block';
        document.getElementById('scanned-gear-name').innerText = decodedText;
        html5QrcodeScanner.clear();
    }
    let html5QrcodeScanner = new Html5QrcodeScanner("reader", { fps: 10, qrbox: 250 });
    html5QrcodeScanner.render(onScanSuccess);

    // --- Update Logic ---
    function updateInventory(action) {
        const gearName = document.getElementById('scanned-gear-name').innerText;
        const id = document.getElementById('memberID').value;
        if(!id) return alert("Enter Room/Member ID");

        let item = inventory.find(i => i.name === gearName);
        
        if (item) {
            if (action === 'issue' && item.issued < item.total) {
                item.issued++;
                alert(`${gearName} issued to ${id}`);
            } else if (action === 'return' && item.issued > 0) {
                item.issued--;
                alert(`${gearName} returned by ${id}`);
            } else {
                alert("Operation not possible (Check stock limits)");
            }
        } else {
            // If gear not in list, add it
            if(action === 'issue') inventory.push({name: gearName, total: 1, issued: 1});
            alert("New gear type detected and logged!");
        }

        renderTable();
        document.getElementById('scan-popup').style.display = 'none';
        html5QrcodeScanner.render(onScanSuccess);
    }

    // --- QR Generator ---
    var qrcode = new QRCode(document.getElementById("qrcode"), { width: 120, height: 120 });
    function generateNewQR() {
        const val = document.getElementById('new-gear-name').value;
        if(!val) return alert("Enter gear name");
        qrcode.makeCode(val);
    }
</script>
